# ngrx-store-effects-assignment

 npm install

 npm install --save @ngrx/store
 
 # Install ngrx/effects
 npm install --save @ngrx/effects
 
 # Install Redux devtools
 npm install --save @ngrx/store-devtools
 
# Build
ng build

# Run
ng server listens localhost:4200
